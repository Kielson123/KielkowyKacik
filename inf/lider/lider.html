Lider


def SzukajLidera(A):
    n = len(A)
    for i in range(n):
        ile = 0
        for j in range(n):
            if A[j] == A[i]:
                ile += 1
        if ile > n % 2:
            return A[i]
    return -1


A = [1, 2, 3, 1, 5, 6, 2, 1, 1, 1, 1]
lider = SzukajLidera(A)
if lider != -1:
    print("Lider:", lider)
else:
    print("Nie istnieje lider w danym zbiorze")
